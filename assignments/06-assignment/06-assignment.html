df_features = pd.read_csv("Data/features.csv")

threshold = 3000

more_weight = df_features.query("weight > @threshold")
less_weight = df_features.query("weight <= @threshold")

mean_more_weight = more_weight["acceleration"].mean()
mean_less_weight = less_weight["acceleration"].mean()

print(f"Mean acceleration for weights over 3000: {mean_more_weight}")
print(f"Mean acceleration for weights under 3000: {mean_less_weight}")
